### 5.2 代码结构

```python
class AIService:
    def __init__(self, customer_tier):
        self.tier = customer_tier
        self.model = self.get_model()
        self.token_limit = self.get_limit()
    
    def generate_reply(self, message, context):
        """生成 AI 回复"""
        if not self.check_limit():
            return self.fallback_reply(message)
        
        prompt = self.build_prompt(message, context)
        response = self.call_ai(prompt)
        self.update_token_count(response.usage)
        
        return response.text
    
    def analyze_sentiment(self, message):
        """情绪分析"""
        prompt = f"分析这句话的情绪：{message}"
        result = self.call_ai(prompt)
        return result.sentiment
    
    def check_limit(self):
        """检查调用限制"""
        return self.used_tokens < self.token_limit
```
